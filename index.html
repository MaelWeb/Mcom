组件使用示例

1.loading

基本调用

$.loading();// 直接传入"show"或是"hide",会分别调用show和hide方法

参数

$.loading({ content:'This is loading text'});// loading 文字，默认是“加载中...”;

示例

2.弹出框

基本调用

$.dialog({
            title: "Dialog",
            content: "This is A Dialog",
            action: function() {
                console.log("action")
            }
    });

参数

$.dialog({
        title: "Dialog", // 标题
        content:'This is loading text', // 内容
        button: {
                action: '确认', // 按钮文字
                dismis: '取消'
            },
        select: 'action', // 高亮的按钮
        dismis: function() {}, // 取消按钮回调
        action: function() {}  // 确认按钮回调
    });

示例

3.提示框

基本调用

$.tips("This is tips text");

参数

$.tips({
        content: '', // 内容
        stayTime: 3000, // 消失时间
        type: 'warning', // 提示类型，可选warning|danger
        callback: function() {} // 提示框消失后回调
    });

示例

4.模态框

模版

<script type="text/template" class="J-modal">
    <div class="modal modal-layout">
        <div class="modal-dialog">
            <div class="modal-content">
                <div class="modal-header">
                    <button type="button" class="close"><span aria-hidden="true">×</span></button>
                    <h4 class="modal-title" id="myModalLabel">Modal title</h4>
                </div>
                <div class="modal-body">
                    <p>模态框主体</p><
                /div>
                <div class="modal-footer">
                    <button type="button" class="btn btn-default J-canlce">Close</button>
                    <button type="button" class="btn btn-primary">Save changes</button>
                </div>
            </div>
        </div>
     </div>
</script>

基本调用

var modal = $.Wimodal($('.J-modal').html(), function() {
        var self = this,
            $this = self.$this;
        $this.find(".J-canlce").click(function() {
            // some code
        });
    });
    modal.show();

参数

$.Wimodal(
        $('.J-modal').html(), // 模态框模版
        function() {
            //something todo
        });


// ps: 也可以通过以下方式调用：
$(".J-modal").Wimodal(function() {
    // something todo
});
